835__STATIC_INLINE void LL_I2C_ConfigSpeed(I2C_TypeDef *I2Cx, uint32_t PeriphClock, uint32_t ClockSpeed,
836 uint32_t DutyCycle)
837{
838 uint32_t freqrange = 0x0U;
839 uint32_t clockconfig = 0x0U;
840
841 /* Compute frequency range */
842 freqrange = __LL_I2C_FREQ_HZ_TO_MHZ(PeriphClock);
843
844 /* Configure I2Cx: Frequency range register */
845 MODIFY_REG(I2Cx->CR2, I2C_CR2_FREQ, freqrange);
846
847 /* Configure I2Cx: Rise Time register */
848 MODIFY_REG(I2Cx->TRISE, I2C_TRISE_TRISE, __LL_I2C_RISE_TIME(freqrange, ClockSpeed));
849
850 /* Configure Speed mode, Duty Cycle and Clock control register value */
851 if (ClockSpeed > LL_I2C_MAX_SPEED_STANDARD)
852 {
853 /* Set Speed mode at fast and duty cycle for Clock Speed request in fast clock range */
854 clockconfig = LL_I2C_CLOCK_SPEED_FAST_MODE | \
855 __LL_I2C_SPEED_FAST_TO_CCR(PeriphClock, ClockSpeed, DutyCycle) | \
856 DutyCycle;
857 }
858 else
859 {
860 /* Set Speed mode at standard for Clock Speed request in standard clock range */
861 clockconfig = LL_I2C_CLOCK_SPEED_STANDARD_MODE | \
862 __LL_I2C_SPEED_STANDARD_TO_CCR(PeriphClock, ClockSpeed);
863 }
864
865 /* Configure I2Cx: Clock control register */
866 MODIFY_REG(I2Cx->CCR, (I2C_CCR_FS | I2C_CCR_DUTY | I2C_CCR_CCR), clockconfig);
867}
